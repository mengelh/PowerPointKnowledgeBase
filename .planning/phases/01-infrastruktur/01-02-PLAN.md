Create ansible/roles/authentik/templates/docker-compose.yml.j2 with the exact structure below.

Critical constraints enforced in this template:
1. authentik-server has NO authentik-forward-auth middleware label (Pitfall 2 prevention)
2. postgresql and valkey are on authentik_internal ONLY — no proxy network attachment
3. authentik-server is on BOTH proxy (for Traefik) and authentik_internal (for DB/Redis)
4. authentik-worker is on authentik_internal ONLY (no routing needed)
5. depends_on uses condition: service_healthy for postgresql and valkey
6. Secrets come from vault vars ({{ vault_authentik_secret_key }}, {{ vault_pg_password }})
7. AUTHENTIK_WEB__PATH set to {{ authentik_web_path }} = /authentik
8. AUTHENTIK_COOKIE_DOMAIN set to {{ authentik_cookie_domain }} = home-engelhardt.uk
9. No version: key (Docker Compose v2 — obsolete key)

```yaml
services:
  postgresql:
    image: docker.io/library/postgres:16-alpine
    container_name: authentik-postgresql
    restart: unless-stopped
    environment:
      POSTGRES_PASSWORD: "{{ vault_pg_password }}"
      POSTGRES_USER: "{{ pg_user }}"
      POSTGRES_DB: "{{ pg_db }}"
    volumes:
      - postgresql_data:/var/lib/postgresql/data
    networks:
      - authentik_internal
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U {{ pg_user }} -d {{ pg_db }}"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 30s

  valkey:
    image: docker.io/valkey/valkey:8-alpine
    container_name: authentik-valkey
    restart: unless-stopped
    command: --save 60 1 --loglevel warning
    volumes:
      - valkey_data:/data
    networks:
      - authentik_internal
    healthcheck:
      test: ["CMD", "valkey-cli", "ping"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 10s

  authentik-server:
    image: {{ authentik_image }}
    container_name: authentik-server
    restart: unless-stopped
    command: server
    environment: &authentik-env
      AUTHENTIK_REDIS__HOST: valkey
      AUTHENTIK_POSTGRESQL__HOST: postgresql
      AUTHENTIK_POSTGRESQL__USER: "{{ pg_user }}"
      AUTHENTIK_POSTGRESQL__NAME: "{{ pg_db }}"
      AUTHENTIK_POSTGRESQL__PASSWORD: "{{ vault_pg_password }}"
      AUTHENTIK_SECRET_KEY: "{{ vault_authentik_secret_key }}"
      AUTHENTIK_WEB__PATH: "{{ authentik_web_path }}"
      AUTHENTIK_COOKIE_DOMAIN: "{{ authentik_cookie_domain }}"
      AUTHENTIK_LISTEN__PROXY_TRUSTED_CIDRS: "172.16.0.0/12,10.0.0.0/8"
    volumes:
      - {{ authentik_compose_dir }}/media:/media
      - {{ authentik_compose_dir }}/custom-templates:/templates
    networks:
      - proxy
      - authentik_internal
    depends_on:
      postgresql:
        condition: service_healthy
      valkey:
        condition: service_healthy
    healthcheck:
      test: ["CMD-SHELL", "wget -qO- http://localhost:9000/-/health/ready/ || exit 1"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 60s
    labels:
      - "traefik.enable=true"
      - "traefik.http.routers.authentik.rule=PathPrefix(`{{ authentik_web_path }}`)"
      - "traefik.http.routers.authentik.entrypoints=websecure"
      - "traefik.http.routers.authentik.tls=true"
      - "traefik.http.services.authentik.loadbalancer.server.port=9000"
      - "traefik.docker.network=proxy"

  authentik-worker:
    image: {{ authentik_image }}
    container_name: authentik-worker
    restart: unless-stopped
    command: worker
    environment: *authentik-env
    user: root
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock
      - {{ authentik_compose_dir }}/media:/media
      - {{ authentik_compose_dir }}/certs:/certs
      - {{ authentik_compose_dir }}/custom-templates:/templates
    networks:
      - authentik_internal
    depends_on:
      postgresql:
        condition: service_healthy
      valkey:
        condition: service_healthy

volumes:
  postgresql_data:
    driver: local
  valkey_data:
    driver: local

networks:
  proxy:
    external: true
    name: proxy
  authentik_internal:
    driver: bridge
    internal: true
```

Important notes for the executor:
- The `authentik-server` labels do NOT include any middleware reference. This is intentional and correct.
  Authentik must be reachable WITHOUT authentication so users can log in (AUTH-03).
- `traefik.docker.network=proxy` tells Traefik which network to use for this multi-network container.
  Without this, Traefik may try to use the internal network and fail to route.
- `AUTHENTIK_LISTEN__PROXY_TRUSTED_CIDRS` tells Authentik to trust X-Forwarded-For headers from
  Docker network ranges — prevents auth redirect URL construction failures.
- `internal: true` on authentik_internal means containers on this network have no external internet
  access — correct for PostgreSQL and Redis isolation.
- Named volumes (postgresql_data, valkey_data) survive container recreation — Ansible will never
  run `docker compose down -v` (Plan 03 verification confirms this).
